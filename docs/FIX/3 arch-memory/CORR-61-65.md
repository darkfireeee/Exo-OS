```rust
// libs/exo-phoenix-ssr/src/lib.rs — AJOUTER des offsets dédiés pour les ACK TLB :

/// Offset du slot ACK Freeze dans la SSR (u32 par core).
pub fn freeze_ack_offset(slot: u32) -> usize {
    SSR_FREEZE_ACK_BASE + (slot as usize) * 4
}

/// Offset du slot ACK TLB dans la SSR (u32 par core).  
/// SÉPARÉ de freeze_ack_offset pour éviter la collision.
pub fn tlb_ack_offset(slot: u32) -> usize {
    SSR_TLB_ACK_BASE + (slot as usize) * 4
}

/// Base des ACK Freeze (zone dédiée, 256 slots × 4 bytes = 1KB).
pub const SSR_FREEZE_ACK_BASE: usize = 0x1000;

/// Base des ACK TLB (zone dédiée, séparée).
pub const SSR_TLB_ACK_BASE: usize = 0x1400;
```

```rust
fn checklist_tlb_shootdown() {
    let cpu_count = crate::arch::x86_64::smp::init::smp_cpu_count() as usize;
    
    // Réinitialiser les slots ACK TLB avant l'IPI.
    for slot in 0..cpu_count {
        unsafe {
            ssr::ssr_atomic_u32(ssr::tlb_ack_offset(slot as u32))
                .store(0, Ordering::Release);
        }
    }
    
    // Envoyer l'IPI TLB broadcast.
    // ... send IPI 0xF3 ...
    
    // Attendre les ACK TLB sur l'offset DÉDIÉ (plus de busy-wait aveugle).
    let deadline_ns = crate::arch::x86_64::time::ktime::ktime_deadline_us(200);

            ssr::ssr_atomic_u32(ssr::tlb_ack_offset(slot as u32))
                .load(Ordering::Acquire) == ssr::TLB_ACK_DONE
        });
        
        if all_acked { break; }
        
        if crate::arch::x86_64::time::ktime::ktime_get_ns() > deadline_ns {

            break;
        }
```

```rust
fn alloc_shadow_stack_pages(count: usize) -> u64 {
    use crate::memory::physical::frame_allocator::{alloc_pages_order, AllocFlags};

    let order = count.next_power_of_two().trailing_zeros() as usize;
    
    match alloc_pages_order(order, AllocFlags::ZEROED | AllocFlags::KERNEL) {
        Some(frame) => {
            let phys = frame.phys_addr().as_u64();
            // Vérification de sécurité : jamais retourner 0.
            if phys == 0 {
                // L'allocateur a retourné la page physique 0 — anormal.
                // Libérer et signaler l'erreur.
                return 0;
            }
            // Leak intentionnel du frame ici — géré par free_shadow_stack_pages().
            core::mem::forget(frame);
            phys
        }
        None => {
            // Allocation échouée — CET ne sera pas activé pour ce thread.
            // L'appelant doit vérifier la valeur de retour.
            0
        }
    }
}

fn free_shadow_stack_pages(base: u64, count: usize) {
    use crate::memory::physical::frame_allocator::free_pages;
    use crate::memory::core::PhysAddr;

    if base == 0 { return; }
    
    let order = count.next_power_of_two().trailing_zeros() as usize;
    // SAFETY: base est une adresse physique allouée par alloc_shadow_stack_pages.
    unsafe {
        free_pages(PhysAddr::new(base), order);
    }
}
```

# CORR-65 — OID placeholder dans ExoLedger (MAJ-03)

**Sévérité :** 🟠 MAJEURE — Audit inutilisable (tous les OIDs = zéros)  
**Fichier :** `kernel/src/security/exoledger.rs:360`

---

## Problème

```rust
let mut oid = [0u8; 32];
// oid[0..8] = pid (placeholder)
// → Tous les lookups audit par OID échouent
```

## Correction

```rust
// exoledger.rs — Remplacer le placeholder par l'extraction réelle depuis CapToken :

fn current_actor_oid() -> [u8; 32] {
    use crate::scheduler::core::tcb::current_tcb;
    use crate::security::capability::token::CapToken;

    // Lire le TCB du thread courant via gs:[0x20] (une fois CORR-GI03 appliqué).
    let tcb = unsafe { current_tcb() };
    
    if let Some(tcb) = tcb {
        // Extraire l'OID depuis le CapToken du thread.
        // Le CapToken est stocké dans le TCB à l'offset cap_token.
        if let Some(token) = tcb.cap_token() {
            return token.owner_oid();
        }
    }
    
    // Fallback : OID basé sur le PID (meilleur que tout-zéros).
    let mut oid = [0u8; 32];
    let pid = unsafe { tcb.map(|t| t.pid()).unwrap_or(0) };
    oid[0..8].copy_from_slice(&pid.to_le_bytes());
    // Distinguer clairement les OIDs "pid-only" des vrais OIDs.
    oid[8] = 0xFF; // marqueur "OID partiel — CapToken non disponible"
    oid
}

// Dans log_event() :
// REMPLACER :
// let mut oid = [0u8; 32];
// AVEC :
let oid = current_actor_oid();
```
